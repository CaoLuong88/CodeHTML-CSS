Bài 03: Khai báo biến kiểu let

Tham khảo mã nguồn:
        <script type="text/javascript">
            // ví dụ về khai báo kiểu let:	
            let studentName_1 = "Hoàng Việt Anh";           
            let check = 4
            if (check > 3) {
                let studentName_2 = "Lê Văn Trung";
                console.log(studentName_1);
                console.log(studentName_2);
            }           
            console.log("========================");
            console.log(studentName_1);		
            // let không cho phép tái cấu trúc nhưng vẫn cho phép thay đổi giá trị.
            console.log("========================");
            studentName_1 = "Trương Việt Hoàng";
            console.log(studentName_1);
            // 2 dòng lệnh phía dưới báo lỗi
            console.log(studentName_2); 
            //let studentName_1 = "Võ Hoàng Bửu";    
        </script>
