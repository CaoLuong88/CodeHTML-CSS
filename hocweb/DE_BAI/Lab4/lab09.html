Bài 09: javascript - Viết hàm và cách gọi hàm

Button - Thẻ A
Tham khảo mã nguồn:
        <div class="row">
            <div class="col-sm-4">
                <input type="button" value="Show log" class="btn btn-primary" onclick="hello_show_log()">
                <input type="button" value="Show alert" class="btn btn-success" onclick="hello_show_alert()">
                <input type="button" value="Show confirm" class="btn btn-danger" onclick="hello_show_confirm()">   
                <input type="button" value="Change form" class="btn btn-secondary" onclick="change_formt()">
                <a href="#" class="btn btn-primary">Button - Thẻ A </a>
            </div>
        </div>
              
        <script type="text/javascript">
            function hello_show_alert() {
                alert("Xin chào");
            }
            function hello_show_confirm(){
                confirm("Hello world");
            }
            function hello_show_log() {
                console.log("Hello world");
            }
            function change_formt() {
                document.location.href = "lab08.html"
            }
        </script>
