Bài 01: Khai báo và hiển thị sử dụng javascript

Tham khảo mã nguồn - tương tác với người dùng trong javascript
        <script type="text/javascript">
            // Chú thích 1 dòng       
            /*Chú thích nhiều dòng*/
            // hiển thị trong Developer-Tool
            console.log("Xin chào ĐH CNTT K25B");
            // hiển thị trong form
            document.write('Xin chào ĐH CNTT K25B');
            // hiển thị vào thẻ html
            document.getElementById("hello").innerHTML ="Xin chào ĐH CNTT K25B";
            // Hiển thị thông báo
            alert("Xin chào ĐH CNTT K25B");
            // Hiển thị thông báo có xác nhận (true hoặc false)
            confirm("Xin chào ĐH CNTT K25B"); 
            // hiển thị thông báo có nhập thông tin và nhận giá trị trả về (giá trị hoặc null)
            prompt("Nhập vào tên của bạn:", "Anonymous");           
        </script>
